\usepackage{fancyhdr,lastpage}
\pagestyle{fancy} \fancyhead{} \renewcommand{\headrulewidth}{0pt}
\usepackage[dvipsnames]{xcolor}

\usepackage{eurosym}

% titlesec tweaking of part titles
% --------------------------------
% (using  titlesec  means we can't use memoir)
\usepackage[compact]{titlesec}

\titleformat{\section}
  {\normalfont      \huge\bfseries\color{BurntOrange}}{\thesection}      {1em}{} % ( Huge = huge at 12pt )
\titleformat{\subsection}
  {\normalfont     \LARGE\bfseries\color{Goldenrod}}  {\thesubsection}   {1em}{}
\titleformat{\subsubsection}
  {\normalfont     \Large\bfseries\color{SpringGreen}}{\thesubsubsection}{1em}{}
% the following also remove  runin:
\titleformat{\paragraph}
  {\normalfont     \large\bfseries\color{SkyBlue}}    {\theparagraph}    {1em}{}
\titleformat{\subparagraph}
  {\normalfont\normalsize\bfseries\color{SkyBlue}}    {\thesubparagraph} {1em}{}

\titlespacing*{\section}       {0pt}{1ex plus 1ex minus .2ex}{0pt}
\titlespacing*{\subsection}    {0pt}{1ex plus 1ex minus .2ex}{0pt}
\titlespacing*{\subsubsection} {0pt}{1ex plus 1ex minus .2ex}{0pt}
\titlespacing*{\paragraph}     {0pt}{1ex plus 1ex minus .2ex}{0pt}
\titlespacing*{\subparagraph}  {0pt}{1ex plus 1ex minus .2ex}{0pt}
