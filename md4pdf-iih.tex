\usepackage{fancyhdr,lastpage} \pagestyle{fancy} \renewcommand{\headrulewidth}{0pt}
\usepackage[dvipsnames]{xcolor}

\usepackage[compact]{titlesec}

% Huge = huge at 12pt, subparagraph formatting is just kept in for occasional overruns
\titleformat{\section}
  {\normalfont      \huge\bfseries\color{Yellow}}     {\thesection}      {1em}{}
\titleformat{\subsection}
  {\normalfont     \LARGE\bfseries\color{Goldenrod}}  {\thesubsection}   {1em}{}
\titleformat{\subsubsection}
  {\normalfont     \Large\bfseries\color{BurntOrange}}{\thesubsubsection}{1em}{}
\titleformat{\paragraph}
  {\normalfont     \large\bfseries\color{BurntOrange}}{\theparagraph}    {1em}{}
\titleformat{\subparagraph}
  {\normalfont\normalsize\bfseries\color{BurntOrange}}{\thesubparagraph} {1em}{}

\titlespacing*{\section}       {0pt}{1ex plus 1ex minus .2ex}{0pt}

\titlespacing*{\subsubsection} {0pt}{1ex plus 1ex minus .2ex}{0pt}
